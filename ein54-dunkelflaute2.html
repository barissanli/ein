repeated_values_w = np.repeat(df14['wind'].values, 4)
repeated_values_s = np.repeat(df14['solar'].values, 4)

new_index = pd.date_range(start='2014-01-01 00:00', periods=len(repeated_values_w), freq='15min')

# Create a new DataFrame with 15-minute intervals
df14_15min = pd.DataFrame({'wind': 1000*repeated_values_w, 'solar':1000*repeated_values_s}, index=new_index)
